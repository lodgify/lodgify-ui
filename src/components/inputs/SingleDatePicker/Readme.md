### Usage

#### Block days

```jsx
<SingleDatePicker
  getIsDayBlocked={moment => moment.format('dddd') === 'Friday'}
/>
```

#### Controlled

```jsx
const Controller = () => {
  const [value, setValue] = React.useState(null);
  return (
    <SingleDatePicker
      onChange={setValue}
      value={value}
      getIsDayBlocked={moment => moment.format('dddd') === 'Friday'}
    />
  );
};
<Controller />;
```
